void do_eh( BYTE n)
{
    BYTE i,j;

    for(i=1;i<n;i++) {
        for(j=0;j < strlen(arg[i]);j++) {
            if(arg[i][j] == '%') {
                switch(arg[i][++j]) {
                    case '%' :
                        putchar('%');
                        break;
                    case  '_':
                        putchar(' ');
                        break;
                    case  'b':
                        putchar('|');
                        break;
                    case  'e':
                        putchar('\x1b');
                        break;
                    case  'g':
                        putchar('>');
                        break;
                    case  'h':
                        putchar('\b');
                        break;
                    case  'l':
                        putchar('<');
                        break;
                    case  'n':
                        putchar('\n');
                        break;
                    case  'r':
                        putchar('\r');
                        break;
                    case  't':
                        putchar('\t');
                        break;
                    default :
                        putchar(arg[i][j]);

                }
            }
            else
                putchar(arg[i][j]);

        }
        printf(" ");
    }
    printf("\n");
}
