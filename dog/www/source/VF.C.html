void do_vf(BYTE n)
{

    if (n==1) {
        printf("Verify is ");
        asm{
            MOV ah,54h  /*  Get verify flag */
            INT 21h
            MOV dl,01h
            CMP al,dl
            JE  on
            }
        puts("OFF");
        return;
        on:
        puts("ON");
        return;
    }
    if (n==2) {
        if (stricmp(arg[1],"ON")==0) {
            asm {
                MOV ah,2eh  /*Set Verify flag*/
                MOV al,01h  /* to on*/
                INT 21
            }
        }
        else if (stricmp(arg[1],"OFF")==0) {
            asm {
                MOV ah,2eh  /* Set Verify flag */
                MOV al,00h  /* to off */
                INT 21
            }
        }
        else {
            puts("You MUST specify either ON or OFF.");
            return;
        }

    }
    if ((n!=1) && (n!=2)) {
        puts("Invalid number of arguments.");
        return;
    }
}
