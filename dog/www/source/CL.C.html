void do_cl(void)
{
    asm{
            MOV ax,0600h
            MOV bh,07h
            MOV cx,0000h        /* (0,0) to*/
            MOV dx,5079h        /* (50,79) */
            INT 10h             /*Clear screen*/
            MOV ah,02h
            MOV bh,00h
            MOV dx,0000h        /* (0,0)*/
            INT 10h             /*Move cursor to upper left corner*/
        }
}
